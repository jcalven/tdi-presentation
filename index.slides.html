# Load hotel data from json file (scraped from hotels.com) and do some light preprocessing.
def read_json(filename):
    """
    Reads json file with hotel data and concatenates the result
    into one dataframe.
    """
    with open(filename, "rb") as f:
        data = json.load(f)

    dfs = [pd.DataFrame(item) for item in data]
    return pd.concat(dfs, ignore_index=True)

def extract_from_strings(df):
    """
    Finds values of interest in strings. Extract, and possibly split into new columns.
    """
    df_ = df.copy()
    price = []
    price_sale = []
    
    # Extract dollar amounts from string
    df_["price"] = df_.price.str.findall(r"(\$\d+)")

    # Extract numeric values in strings
    df_["rating"] = df_.rating.str.extract(r"(\d+\.*\d*)").astype(float)
    df_["n_reviews"] = df_.n_reviews.str.replace(",","").str.extract(r"(\d+)").astype(int)
    df_["star_rating"] = df_.star_rating.str.extract(r"(\d+\.*\d*)").astype(float)
    
    for row_ in df_.iterrows():
        # Split dollar amounts into two columns (`price`and price_sale`)
        row = row_[1]["price"]
        if not row:
            price.append(None)
            price_sale.append(None)
        elif len(row) == 1:
            price.append(int(row[0].lstrip("$")))
            price_sale.append(None)
        elif len(row) == 2:
            price.append(int(row[0].lstrip("$")))
            price_sale.append(int(row[1].lstrip("$")))
        
    df_["price"] = price
    df_["price_sale"] = price_sale
    return df_

def preprocess(df):
    """
    Runs a set of preprocessing actions. 
    """
    
    # Drop columns
    drop_cols = ["adults", "children", "address", "review_box"]

    # Cast date columns as datetime
    df["checkin"] = pd.to_datetime(df["checkin"])
    df["checkout"] = pd.to_datetime(df["checkout"])
    df = df.drop(columns=drop_cols)
    
    # Extract data from string fields
    df = extract_from_strings(df)
    return df

def n_lowest_prices(df, column, n=3):
    return df.nsmallest(n, column)

def display_side_by_side(dfs, captions, padding=1):
    """Display tables side by side to save vertical space
    Input:
        dfs: list of pandas.DataFrame
        captions: list of table captions
    """
    output = ""
    combined = dict(zip(captions, dfs))
    for caption, df in combined.items():
        output += df.style.set_table_attributes("style='display:inline'").set_caption(caption)._repr_html_()
        output += padding * "\xa0"
    display(HTML(output))

class GridPlot(object):
    """
    Helper class for plotting multiple datasets superimposed and/or as subplots in a grid.
    
    Args:
        data (dict): Data dictionary with optional plotting options.
        grid (tuple, optional): Tuple indicating figure grid size. Defaults to (1,).
            
    Example:
        x = [0,1,2,3,4,5]
        y1 = [10,8,5,6,9,12]
        y2 = [9,6,4,5,2,1]
        X = (x, x)
        X = (y1, y2)
        data = dict(
            # Figure that will appear in place 0 in the figure grid
            0 = {"x": X,
                "y": Y, 
                "plot_options": [
                    {"label": "label of y1 data"},
                    {"label": "label of y2 data"},
                    ],
                "ax_options": {
                    "legend": True, 
                    "ylabel": "y-axis name", 
                    "xlabel": "x-axis name", 
                    "title": "figure title"}
                },
            # Figure that will appear in place 1 in the figure grid
            1 = {"x": x, 
                "y": y1, 
                "plot_options": [
                    {"label": "label of y1 data"},
                    ],
                "ax_options": {
                    "legend": True, 
                    "ylabel": "y-axis name", 
                    "xlabel": "x-axis name", 
                    "title": "figure title"}
                },
        gp = GridPlot()
        gp.plot()
    """
    
    def __init__(self, data, grid=(1,1)):
        
        self.data = data
        self.axgrid = grid
        assert len(self.data) == sum(self.axgrid), f"data must be of size n_plots ({len(data)} != {self.axgrid})"

    def plot(self, figsize=(8,6), sharex=False, sharey=False):
        """
        Creates the plots.
        
        Args:
            figsize (tuple): Tuple of size 2 indicating the figure size. Defaults to (8,6).
        """
        
        # Set up the axes with gridspec
        fig, axes = plt.subplots(*self.axgrid, figsize=figsize, sharex=sharex, sharey=sharey)
        
        if not isinstance(axes, np.ndarray):
            axes = [axes]
        else:
            axes = axes.flatten()
        for i, ax in enumerate(axes):
            data = self.data.get(i)
            x = data.get("x")
            y = data.get("y")
            plot_options = data.get("plot_options")
            ax_options = data.get("ax_options")

            if not isinstance(x, (tuple)):
                x = tuple([x])
            if not isinstance(y, (tuple)):
                y = tuple([y])
            if not isinstance(plot_options, (list)):
                plot_options = [plot_options]
                
            for xvals, yvals, plot_opts in zip(x, y, plot_options):
                ax.plot(xvals, yvals, **plot_opts)

            label_size = ax_options.get("label_size", 12)
            
            if "legend" in ax_options:
                ax.legend()
            if "title" in ax_options:
                ax.set_title(ax_options.get("title"), size=label_size)
            if "xlabel" in ax_options:
                ax.set_xlabel(ax_options.get("xlabel"), size=label_size)
            if "ylabel" in ax_options:
                ax.set_ylabel(ax_options.get("ylabel"), size=label_size)
        fig.tight_layout()

Average prices¶

Let's start with plotting the average hotel prices for each day and for all hotels. Note that these are the prices on each day between 2020-05-05 and 2020-11-05 as they were seen on 2020-05-04 on https://hotels.com.

We can see that the price drops significantly approximately 1 and 6 months from the search date, while the general trend is an increasing price the further from the search date the hotel is booked on. The high-frequency fluctuations can likely be attributed to which day of the week the hotel is booked for.

Star rating¶

The pricing strategy for 2, 3, 4 and 5-star hotels may differ due to targeting different customer segments.

Splitting the hotels into subsets of 2, 3, 4 and 5-star hotels and plotting their average price curves, we can see that this indeed seem to be the case.

  • 2-star: Almost flat curve, indicating no major significance in pricing when booked early or late.
  • 3-star: Close to the characteristic of the plot above. Price tend to increase with time.
  • 4-star: Similar to 3-star hotels.
  • 5-star: Non-trivial trend. High prices when booked early and late, with lower prices in between.

Seasonal effect¶

What season it is may play a role in the pricing of hotels, and to get a more accurate picture of the actual change in price we must then first correct the prices by removing the seasonal dependence.

Using the 2018 Trivago Hotel Price Index1 for Las Vegas, we can get an idea of the seasonality effect on the price. Note that the index prices are only for one year. Ideally, we should use price indexes over multiple years to make sure 2018 was not special in some way. However, short of price indexes in other years, the 2018 data will have to suffice with the caveat that it might not be representative of the real seasonal hotel price fluctuations in Las Vegas. Since we're not building a predictive model just yet, this should not pose a big problem.

Conclusion¶

With the exception of the 2-star hotels, the 3 lowest prices occur at different times depending on if we use the uncorrected price or the corrected price. In particular,

  • 2-star: The 3 lowest prices are the same for the uncorrected and corrected price, and occur approximately 2 and 6 months from the search date.
  • 3-star: For the corrected prices, the 2 lowest prices are clustered approximately 6 months from the search date, with the 3rd lowest price occuring approximately one month from the search date. The uncorrected prices are split between 6 and 2 months from the search date.
  • 4-star: Using the corrected prices, hotels are cheapest 1 month from search date while the lowest uncorrected prices are split between early 2 and 1 month from the search date, respectively.
  • 5-star: Cheapest corrected prices are clustered approximately 6 months from the search date, while the lowest uncorrected prices are split between approximatly 6 and 2 months from the search date.

Hotel prices are dynamic in nature, determined by factors such as pricing strategies, demand, season, day of week, etc. Being able to accurately estimate the optimal timing for booking a hotel room has the potential to save a lot of money for consumers. In this study we have just looked at prices originating from one search date, and while limited in scope, we can already see that optimal pricing differs among hotels in different star rating categories, and can vary by as much as 6 months (the upper limit of our search range). This shows that it can be more advantageous to book a hotel 6 months in advance as opposed to last-minute, and vice versa. This project will try to model prices and with statistical certainty determine the optimal time to book.

Here we look at other temporal effects that may influence the result.

The weekday effect¶

A closer look at the data reveals another temporal structure: the weekday effect. The figure below show that the price typically spikes on Sundays and Mondays in the 2, 3 and 4-star categories. The effect is not as regular for 5-star hotels. Just like we did with the seasonal effect, we can try to compensate for this effect.
